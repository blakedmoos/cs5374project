```python
from mnemonic import Mnemonic

def check_bip39_words(word_list):
    """
    Checks a list of words against the BIP-39 English word list.
    
    Args:
        word_list (list): A list of strings representing the words to check.
        
    Returns:
        dict: A dictionary mapping each word to a boolean indicating if it is valid.
    """
    # Initialize the Mnemonic class with the English wordlist
    mnemo = Mnemonic("english")
    
    # The wordlist is stored as a tuple in mnemo.wordlist
    bip39_english_words = set(mnemo.wordlist) 
    
    results = {}
    for word in word_list:
        # Clean the input word (lowercase and strip whitespace) just in case
        clean_word = str(word).strip().lower()
        results[clean_word] = clean_word in bip39_english_words
        
    return results

# --- Example Usage ---
if __name__ == "__main__":

    validation_results = check_bip39_words(test_words)
```

Here is the full implementation of how a BIP-39 checksum is calculated and verified from scratch. 

To understand the code, it helps to know the math behind a 12-word BIP-39 phrase:
1. The BIP-39 wordlist has 2,048 words. This means each word represents exactly **11 bits** of data ($2^{11} = 2048$).
2. A 12-word phrase equals **132 bits** total ($12 \times 11$).
3. The first **128 bits** represent the actual random entropy.
4. The remaining **4 bits** represent the checksum.
5. The checksum is calculated by taking the SHA-256 hash of the 128-bit entropy and extracting the first 4 bits.

Here is the Python code to perform this validation:

```python
import hashlib

def validate_bip39_12_word_phrase(mnemonic_phrase, wordlist):
    """
    Validates a 12-word BIP-39 mnemonic phrase by checking its SHA-256 checksum.
    
    Args:
        mnemonic_phrase (str): A space-separated 12-word string.
        wordlist (list): A list of the 2048 valid BIP-39 English words.
        
    Returns:
        bool: True if valid, False otherwise.
        str: A message explaining the result.
    """
    words = mnemonic_phrase.strip().lower().split()
    
    if len(words) != 12:
        return False, f"Expected 12 words, got {len(words)}."

    # Create a dictionary for fast O(1) index lookups
    word_to_index = {word: i for i, word in enumerate(wordlist)}

    # 1. Convert each word to its 11-bit binary representation
    binary_string = ""
    for word in words:
        if word not in word_to_index:
            return False, f"Invalid word found: '{word}' is not in the BIP-39 wordlist."
        
        # Get the index (0-2047) and format it as an 11-bit binary string
        index = word_to_index[word]
        binary_string += f"{index:011b}"

    # 2. Split the 132 bits into 128 bits of entropy and 4 bits of checksum
    entropy_bits = binary_string[:128]
    expected_checksum_bits = binary_string[128:]

    # 3. Convert the 128-bit entropy string into bytes (16 bytes)
    entropy_int = int(entropy_bits, 2)
    entropy_bytes = entropy_int.to_bytes(16, byteorder='big')

    # 4. Compute the SHA-256 hash of the entropy bytes
    hash_bytes = hashlib.sha256(entropy_bytes).digest()

    # 5. Convert the first byte of the hash to binary and extract the first 4 bits
    # (We only need the first byte because our checksum is only 4 bits long)
    first_hash_byte_binary = f"{hash_bytes[0]:08b}"
    actual_checksum_bits = first_hash_byte_binary[:4]

    # 6. Compare the expected checksum with the actual calculated checksum
    if expected_checksum_bits == actual_checksum_bits:
        return True, "Valid mnemonic checksum."
    else:
        return False, "Invalid checksum. The phrase contains a typo or is not a valid BIP-39 mnemonic."
```

### How to adapt this for 15, 18, 21, or 24 words:

* **24 words:** 264 bits / 33 = 8 bits checksum (1 full byte)

You would just dynamically calculate `checksum_length = len(binary_string) // 33` and split the string accordingly!
