```python
def validate_bip39_words(word_list):
```

Here is the full implementation. To understand how the checksum works for a 12-word BIP-39 phrase, it helps to look at the math behind it:

1. The BIP-39 English wordlist has 2,048 words.
2. Because $2^{11} = 2048$, each word represents exactly **11 bits** of data (its index from 0 to 2047).
3. A 12-word phrase equals **132 bits** total (12 words × 11 bits).
4. These 132 bits are split into two parts: 
   * **128 bits** of actual entropy (the random seed data).
   * **4 bits** of checksum.
5. The checksum is calculated by taking the SHA-256 hash of the 128-bit entropy and extracting the first 4 bits.

Here is the Python code that performs this exact extraction and validation from scratch, using only standard libraries:

```python
import hashlib
import urllib.request

def get_bip39_wordlist():
    """Fetches the official BIP-39 English wordlist."""
    url = "https://raw.githubusercontent.com/bitcoin/bips/master/bip-0039/english.txt"
    with urllib.request.urlopen(url) as response:
        return response.read().decode('utf-8').splitlines()

def validate_12_word_checksum(mnemonic_phrase, wordlist=None):
    """
    Validates the checksum of a 12-word BIP-39 mnemonic phrase.
    Returns a tuple: (is_valid: bool, message: str)
    """
    if wordlist is None:
        wordlist = get_bip39_wordlist()
        
    words = mnemonic_phrase.strip().lower().split()
    
    if len(words) != 12:
        return False, f"Expected 12 words, got {len(words)}."
        
    # 1. Convert words to their 11-bit binary index
    binary_string = ""
    for word in words:
        try:
            index = wordlist.index(word)
            # Format the index as an 11-bit binary string (e.g., 5 -> '00000000101')
            binary_string += f"{index:011b}"
        except ValueError:
            return False, f"Word '{word}' is not in the BIP-39 wordlist."
            
    # 2. Split the 132 bits into 128 bits of entropy and 4 bits of checksum
    entropy_bits = binary_string[:128]
    provided_checksum_bits = binary_string[128:]
    
    # 3. Convert the 128-bit entropy string into exactly 16 bytes
    # int(entropy_bits, 2) converts base-2 string to an integer
    entropy_bytes = int(entropy_bits, 2).to_bytes(16, byteorder='big')
    
    # 4. Hash the entropy bytes using SHA-256
    hash_bytes = hashlib.sha256(entropy_bytes).digest()
    
    # 5. Extract the first 4 bits of the hash
    # hash_bytes[0] is the first byte (integer 0-255). 
    # We convert it to an 8-bit string and slice the first 4 characters.
    first_hash_byte_binary = f"{hash_bytes[0]:08b}"
    calculated_checksum_bits = first_hash_byte_binary[:4]
    
    # 6. Compare the calculated checksum with the provided checksum
    if calculated_checksum_bits == provided_checksum_bits:
        return True, "Valid mnemonic! Checksum matches."
    else:
        return False, (f"Invalid checksum! "
                       f"Expected '{calculated_checksum_bits}', "
                       f"but got '{provided_checksum_bits}'.")
```

### How the bit manipulation works in Python:
* `f"{index:011b}"`: This is Python f-string formatting. The `011` means "pad with zeros to a length of 11", and the `b` means "convert the integer to binary".
* `int(entropy_bits, 2)`: Parses the string of 1s and 0s back into a standard Python integer.
* `.to_bytes(16, byteorder='big')`: Converts that giant integer into a 16-byte array, which is required by `hashlib.sha256()`.

```python
def complete_12_word_phrase(eleven_word_stem):
    """
    Takes 11 BIP-39 words and finds a valid 12th word to satisfy the SHA-256 checksum.
    Returns a list of valid 12-word phrases (there will be exactly 128 valid options).
    """
    wordlist = get_bip39_wordlist()
    words = eleven_word_stem.strip().lower().split()
    
    if len(words) != 11:
```
